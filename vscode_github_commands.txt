# Commands to connect Visual Studio Code to GitHub

# 1) Install GitHub CLI (gh) - Debian/Ubuntu example
sudo apt update
sudo apt install gh -y

# Alternative using snap
# sudo snap install gh --classic

# 2) Configure git identity
git config --global user.name "Your Name"
git config --global user.email "you@example.com"

# 3) Authenticate with GitHub (opens browser)
gh auth login --web

# 4) Verify auth status
gh auth status

# 5) Optional: generate an SSH key and add it to GitHub
ssh-keygen -t ed25519 -C "you@example.com"
# then upload the public key with gh
gh ssh-key add ~/.ssh/id_ed25519.pub --title "VSCode-$(hostname)"

# 6) Optional: set git remote URLs to use SSH (example)
# git remote set-url origin git@github.com:USERNAME/REPO.git

# Notes:
# - Open VS Code and use the Command Palette (Ctrl+Shift+P) -> "Sign in to GitHub" to sign in from the editor.
# - If you ran `gh auth login`, VS Code often detects the existing auth automatically.
# - To sync settings, enable Settings Sync in VS Code and sign in with GitHub when prompted.
